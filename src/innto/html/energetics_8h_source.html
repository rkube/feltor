86 {

177 std::vector<double> Energetics<n>::exb_energies(const Matrix<double, TL_DFT>& potential )
178 {
179  std::vector<double> energies;
180  double norm = alg.h*alg.h/(double)(rows*cols); //h^2 from integration and 1/NxNy from complex scalar product
181  phi_[0] = potential;
182  dft_dft.r2c( phi_[0], cphi_[0]);
183  for( size_t k=1; k<n; k++)
184  {
185 #pragma omp parallel for
186  for( size_t i = 0; i < crows; i++)
187  for( size_t j = 0; j < ccols; j++)
188  cphi_[k](i,j) = (a_mu_gamma0_coeff[k-1](i,j))*cphi_[0](i,j);
189  energies.push_back( 0.5*dft_dft.dot( cphi_[0], cphi_[k])*norm);
190  }
191  //averages
192  double norm_avg = alg.h*alg.h/(double)(rows)/(double)cols; //h*h from integration 1/Ny from average and 1/Nx from complex scalar product
193  std::vector<complex> sum_phi[n];
194  for( unsigned i=0; i<n; i++)
195  sum_phi[i] = extract_sum_y( cphi_[i]);
196  for( size_t k=1; k<n; k++)
197  energies.push_back( 0.5*dot( sum_phi[0], sum_phi[k])*norm_avg);
198

283 template<size_t n>
284 std::vector<double> Energetics<n>::diffusion( const std::array<Matrix<double, TL_DFT>, n>& density , const std::array<Matrix<double, TL_DFT>, n>& potential)
285 {
286  //compute -nu*laplace^2 density in dens_
287  dens_ = density;
288 #pragma omp parallel for
289  for( unsigned k=0; k<n; k++)
290  {
291  dft_dft.r2c( dens_[k], cdens_[k]);
292  for( size_t i = 0; i < crows; i++)
293  for( size_t j = 0; j < ccols; j++)
294  cdens_[k](i,j) = diff_coeff(i,j)*cdens_[k](i,j)/(double)(rows*cols);
295  dft_dft.c2r( cdens_[k], dens_[k]);
296  }
297  //add up diffusion
298  std::vector<double> diffusion_(2, 0.);
299  for( unsigned i=0; i<n; i++)
300  {
301  diffusion_[0] += a_[i]*dot( potential[i], dens_[i])*alg.h*alg.h;
302  diffusion_[0] += a_[i]*tau_[i]*dot( density[i], dens_[i])*alg.h*alg.h;
303  }
304  //compute mean diffusion

307  std::vector<double> sum_dens = extract_sum_y( dens_[k] );
308  std::vector<double> sum_phi = extract_sum_y( potential[k] );
309  diffusion_[1] += a_[k]*dot( sum_dens, sum_phi)*alg.h*alg.h/(double)(rows);
310  }
