238 {
239 #ifdef TL_DEBUG
240  if( c_origin.isVoid())
241  throw Message( "Init your coefficients first!", _ping_);
242 #endif
243  //invert coefficients
244  for(unsigned i=0; i<c_inv.rows(); i++)
245  for( unsigned j=0; j<c_inv.cols(); j++)
246  {
247  for( unsigned k=0; k<n; k++)
248  {
249  for( unsigned q=0; q<n; q++)
250  c_inv(i,j)(k,q) = -prefactor*dt*c_origin(i,j)(k,q);
251  c_inv(i,j)(k,k) += prefactor*Coefficients<S>::gamma_0;
252 
253  }
254  invert( c_inv(i,j), c_inv(i,j));

262 void Karniadakis<n,T,P>::step_i( std::array< Matrix<double, P>, n>& v0, std::array< Matrix<double, P>, n> & n0)
263 {
264  for( unsigned k=0; k<n; k++)
265  {
266 #ifdef TL_DEBUG
267  if( v0[k].isVoid()||n0[k].isVoid())
268  throw Message( "ERROR: Cannot work on void matrices!\n", _ping_);
269  if( v0[k].rows() != rows || v0[k].cols() != cols)
270  throw Message( "ERROR: One of the v0 has wrong size!\n", _ping_);
271  if( n0[k].rows() != rows || n0[k].cols() != cols)
272  throw Message( "ERROR: One of the n0 has wrong size!\n", _ping_);
273 #endif
274 #pragma omp parallel for
275  for( size_t i = 0; i < rows; i++)
276  {
277  for( size_t j = 0; j < cols; j++)
278  {
279  n2[k](i,j) = Coefficients<S>::alpha[0]*v0[k](i,j)
280  + Coefficients<S>::alpha[1]*v1[k](i,j)
281  + Coefficients<S>::alpha[2]*v2[k](i,j)
282  + dt*( Coefficients<S>::beta[0]*n0[k](i,j)
283  + Coefficients<S>::beta[1]*n1[k](i,j)
284  + Coefficients<S>::beta[2]*n2[k](i,j));
285  }
286  }
287  swap_fields( n2[k], v2[k]); //we want to keep v2 not n2
288 
289  permute_fields( n0[k], n1[k], n2[k]);
290  permute_fields( v0[k], v1[k], v2[k]);
291  }
292 }
